151ComputeAdtQcMetricsResults<Sum_, Detected_> compute_adt_qc_metrics(

156 auto NC = mat.ncol();

174 compute_adt_qc_metrics(mat, subsets, x, options);

206template<typename Float_, class Host_, typename Sum_, typename Detected_, typename BlockSource_>
207void adt_populate(Host_& host, size_t n, const ComputeAdtQcMetricsBuffers<Sum_, Detected_>& res, BlockSource_ block, const ComputeAdtQcFiltersOptions& options) {
208 constexpr bool unblocked = std::is_same<BlockSource_, std::false_type>::value;
209 constexpr bool oneblocked = std::is_same<BlockSource_, std::true_type>::value;
210
211 auto buffer = [&]() {
212 if constexpr(unblocked || oneblocked) {
213 return std::vector<Float_>(n);
214 } else {
215 return FindMedianMadWorkspace<Float_, size_t>(n, block);
216 }
217 }();
218
219 {
220 ChooseFilterThresholdsOptions opts;
221 opts.num_mads = options.detected_num_mads;
222 opts.log = true;
223 opts.upper = false;
224 opts.min_diff = -std::log(1 - options.detected_min_drop);
225 host.get_detected() = [&]() {
226 if constexpr(unblocked || oneblocked) {
227 auto thresh = choose_filter_thresholds(n, res.detected, buffer.data(), opts).lower;
228 return maybe_vector<oneblocked>(thresh);
229 } else {
230 return internal::strip_threshold<true>(choose_filter_thresholds_blocked(n, res.detected, block, &buffer, opts));
231 }
232 }();
233 }
234
235 {
236 size_t nsubsets = res.subset_sum.size();
237 host.get_subset_sum().resize(nsubsets);
238
239 ChooseFilterThresholdsOptions opts;
240 opts.num_mads = options.subset_sum_num_mads;
241 opts.log = true;
242 opts.lower = false;
243
244 for (size_t s = 0; s < nsubsets; ++s) {
245 auto sub = res.subset_sum[s];
246 host.get_subset_sum()[s] = [&]() {
247 if constexpr(unblocked || oneblocked) {
248 auto thresh = choose_filter_thresholds(n, sub, buffer.data(), opts).upper;
249 return maybe_vector<oneblocked>(thresh);
250 } else {
251 return internal::strip_threshold<false>(choose_filter_thresholds_blocked(n, sub, block, &buffer, opts));
252 }
253 }();
254 }
255 }

258template<class Host_, typename Sum_, typename Detected_, typename BlockSource_, typename Output_>
259void adt_filter(const Host_& host, size_t n, const ComputeAdtQcMetricsBuffers<Sum_, Detected_>& metrics, BlockSource_ block, Output_* output) {
260 std::fill_n(output, n, 1);
261
262 const auto& detected = host.get_detected();
263 for (size_t i = 0; i < n; ++i) {
264 output[i] = output[i] && (metrics.detected[i] >= define_threshold(detected, block, i));
265 }
266
267 size_t nsubsets = metrics.subset_sum.size();
268 for (size_t s = 0; s < nsubsets; ++s) {
269 auto sub = metrics.subset_sum[s];
270 const auto& sthresh = host.get_subset_sum()[s];
271 for (size_t i = 0; i < n; ++i) {
272 output[i] = output[i] && (sub[i] <= define_threshold(sthresh, block, i));
273 }
274 }
275}
