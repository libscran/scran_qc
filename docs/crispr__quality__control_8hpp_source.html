202template<typename Float_, class Host_, typename Sum_, typename Detected_, typename Value_, typename Index_, typename BlockSource_>
203void crispr_populate(Host_& host, size_t n, const ComputeCrisprQcMetricsBuffers<Sum_, Detected_, Value_, Index_>& res, BlockSource_ block, const ComputeCrisprQcFiltersOptions& options) {
204 constexpr bool unblocked = std::is_same<BlockSource_, std::false_type>::value;
205 constexpr bool oneblocked = std::is_same<BlockSource_, std::true_type>::value;
206
207 auto buffer = [&]() {
208 if constexpr(unblocked || oneblocked) {
209 return std::vector<Float_>(n);
210 } else {
211 return FindMedianMadWorkspace<Float_, size_t>(n, block);
212 }
213 }();
214
215 // Subsetting to the observations in the top 50% of proportions.
216 static_assert(std::is_floating_point<Float_>::value);
217 std::vector<Float_> maxprop;
218 maxprop.reserve(n);
219 for (size_t i = 0; i < n; ++i) {
220 maxprop.push_back(static_cast<Float_>(res.max_value[i]) / static_cast<Float_>(res.sum[i]));
221 }
222
223 FindMedianMadOptions fopt;
224 fopt.median_only = true;
225 auto prop_res = [&]() {
226 if constexpr(unblocked || oneblocked) {
227 return find_median_mad(n, maxprop.data(), buffer.data(), fopt);
228 } else {
229 return find_median_mad_blocked(n, maxprop.data(), block, &buffer, fopt);
230 }
231 }();
232
233 for (size_t i = 0; i < n; ++i) {
234 auto limit = [&]() {
235 if constexpr(unblocked || oneblocked){
236 return prop_res.median;
237 } else {
238 return prop_res[block[i]].median;
239 }
240 }();
241 if (maxprop[i] >= limit) {
242 maxprop[i] = res.max_value[i];
243 } else {
244 maxprop[i] = std::numeric_limits<Float_>::quiet_NaN(); // ignored during threshold calculation.
245 }
246 }
247
248 // Filtering on the max counts.
249 ChooseFilterThresholdsOptions copt;
250 copt.num_mads = options.max_value_num_mads;
251 copt.log = true;
252 copt.upper = false;
253 host.get_max_value() = [&]() {
254 if constexpr(unblocked || oneblocked) {
255 auto thresh = choose_filter_thresholds(n, maxprop.data(), buffer.data(), copt).lower;
256 return maybe_vector<oneblocked>(thresh);
257 } else {
258 return internal::strip_threshold<true>(choose_filter_thresholds_blocked(n, maxprop.data(), block, &buffer, copt));
259 }
260 }();
261}

263template<class Host_, typename Sum_, typename Detected_, typename Value_, typename Index_, typename BlockSource_, typename Output_>
264void crispr_filter(const Host_& host, size_t n, const ComputeCrisprQcMetricsBuffers<Sum_, Detected_, Value_, Index_>& metrics, BlockSource_ block, Output_* output) {
265 std::fill_n(output, n, 1);
266
267 const auto& mv = host.get_max_value();
268 for (size_t i = 0; i < n; ++i) {
269 output[i] = output[i] && (metrics.max_value[i] >= define_threshold(mv, block, i));
270 }
271}
