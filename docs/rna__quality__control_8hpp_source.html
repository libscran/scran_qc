91template<typename Value_, typename Index_, typename Subset_, typename Sum_, typename Detected_, typename Proportion_>
92void compute_rna_qc_metrics(
93 const tatami::Matrix<Value_, Index_>& mat,
94 const std::vector<Subset_>& subsets,
95 const ComputeRnaQcMetricsBuffers<Sum_, Detected_, Proportion_>& output,
96 const ComputeRnaQcMetricsOptions& options)
97{
98 auto NC = mat.ncol();
99 size_t nsubsets = subsets.size();
100
101 PerCellQcMetricsBuffers<Sum_, Detected_, Value_, Index_> tmp;
102 tmp.sum = output.sum;
103 tmp.detected = output.detected;
104
105 constexpr bool same_type = std::is_same<Sum_, Proportion_>::value;
106 typename std::conditional<same_type, bool, std::vector<std::vector<Sum_> > >::type placeholder_subset;
107
108 if (output.subset_proportion.size()) {
109 // Providing space for the subset sums if they're not the same type.
110 if constexpr(same_type) {
111 tmp.subset_sum = output.subset_proportion;
112 } else {
113 placeholder_subset.resize(nsubsets);
114 tmp.subset_sum.resize(nsubsets);
115 for (size_t s = 0; s < nsubsets; ++s) {
116 auto& b = placeholder_subset[s];
117 b.resize(NC);
118 tmp.subset_sum[s] = b.data();
119 }
120 }
121 }
122
123 PerCellQcMetricsOptions opt;
124 opt.num_threads = options.num_threads;
125 per_cell_qc_metrics(mat, subsets, tmp, opt);
126
127 for (size_t s = 0 ; s < nsubsets; ++s) {
128 auto dest = output.subset_proportion[s];
129 if (dest) {
130 auto src = tmp.subset_sum[s];
131#ifdef _OPENMP
132 #pragma omp simd
133#endif
134 for (Index_ c = 0; c < NC; ++c) {
135 dest[c] = static_cast<Proportion_>(src[c]) / static_cast<Proportion_>(tmp.sum[c]);
136 }
137 }
138 }
139}

242 return std::vector<Float_>(n);

246 }();

272 }
273 }();
274 }

296template<class Host_, typename Sum_, typename Detected_, typename Proportion_, typename BlockSource_, typename Output_>
297void rna_filter(const Host_& host, size_t n, const ComputeRnaQcMetricsBuffers<Sum_, Detected_, Proportion_>& metrics, BlockSource_ block, Output_* output) {
298 constexpr bool unblocked = std::is_same<BlockSource_, bool>::value;
299 std::fill_n(output, n, 1);
300
301#ifdef _OPENMP
302 #pragma omp simd
303#endif
304 for (size_t i = 0; i < n; ++i) {
305 auto thresh = [&]() {
306 if constexpr(unblocked) {
307 return host.get_sum();
308 } else {
309 return host.get_sum()[block[i]];
310 }
311 }();
312 output[i] = output[i] && (metrics.sum[i] >= thresh);
313 }
314
315#ifdef _OPENMP
316 #pragma omp simd
317#endif
318 for (size_t i = 0; i < n; ++i) {
319 auto thresh = [&]() {
320 if constexpr(unblocked) {
321 return host.get_detected();
322 } else {
323 return host.get_detected()[block[i]];
324 }
325 }();
326 output[i] = output[i] && (metrics.detected[i] >= thresh);
327 }
328
329 size_t nsubsets = metrics.subset_proportion.size();
330 for (size_t s = 0; s < nsubsets; ++s) {
331 auto sub = metrics.subset_proportion[s];
332 const auto& sthresh = host.get_subset_proportion()[s];
333#ifdef _OPENMP
334 #pragma omp simd
335#endif
336 for (size_t i = 0; i < n; ++i) {
337 auto thresh = [&]() {
338 if constexpr(unblocked) {
339 return sthresh;
340 } else {
341 return sthresh[block[i]];
342 }
343 }();
344 output[i] = output[i] && (sub[i] <= thresh);
345 }
346 }
347}
