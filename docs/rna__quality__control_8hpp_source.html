91template<typename Value_, typename Index_, typename Subset_, typename Sum_, typename Detected_, typename Proportion_>
92void compute_rna_qc_metrics(
93 const tatami::Matrix<Value_, Index_>& mat,
94 const std::vector<Subset_>& subsets,
95 const ComputeRnaQcMetricsBuffers<Sum_, Detected_, Proportion_>& output,
96 const ComputeRnaQcMetricsOptions& options)
97{
98 auto NC = mat.ncol();
99 size_t nsubsets = subsets.size();
100
101 PerCellQcMetricsBuffers<Sum_, Detected_, Value_, Index_> tmp;
102 tmp.sum = output.sum;
103 tmp.detected = output.detected;
104
105 constexpr bool same_type = std::is_same<Sum_, Proportion_>::value;
106 typename std::conditional<same_type, bool, std::vector<std::vector<Sum_> > >::type placeholder_subset;
107
108 if (output.subset_proportion.size()) {
109 // Providing space for the subset sums if they're not the same type.
110 if constexpr(same_type) {
111 tmp.subset_sum = output.subset_proportion;
112 } else {
113 placeholder_subset.resize(nsubsets);
114 tmp.subset_sum.resize(nsubsets);
115 for (size_t s = 0; s < nsubsets; ++s) {
116 auto& b = placeholder_subset[s];
117 b.resize(NC);
118 tmp.subset_sum[s] = b.data();
119 }
120 }
121 }
122
123 PerCellQcMetricsOptions opt;
124 opt.num_threads = options.num_threads;
125 per_cell_qc_metrics(mat, subsets, tmp, opt);
126
127 for (size_t s = 0 ; s < nsubsets; ++s) {
128 auto dest = output.subset_proportion[s];
129 if (dest) {
130 auto src = tmp.subset_sum[s];
131 for (Index_ c = 0; c < NC; ++c) {
132 dest[c] = static_cast<Proportion_>(src[c]) / static_cast<Proportion_>(tmp.sum[c]);
133 }
134 }
135 }
136}

235void rna_populate(Host_& host, size_t n, const ComputeRnaQcMetricsBuffers<Sum_, Detected_, Proportion_>& res, BlockSource_ block, const ComputeRnaQcFiltersOptions& options) {
236 constexpr bool unblocked = std::is_same<BlockSource_, bool>::value;
237 auto buffer = [&]() {
238 if constexpr(unblocked) {
239 return std::vector<Float_>(n);

241 return FindMedianMadWorkspace<Float_, size_t>(n, block);

243 }();

245 {
246 ChooseFilterThresholdsOptions opts;

251 if constexpr(unblocked) {
252 return choose_filter_thresholds(n, res.sum, buffer.data(), opts).lower;
253 } else {

255 }
256 }();
257 }

262 opts.log = true;

269 }

271 }

293template<class Host_, typename Sum_, typename Detected_, typename Proportion_, typename BlockSource_, typename Output_>
294void rna_filter(const Host_& host, size_t n, const ComputeRnaQcMetricsBuffers<Sum_, Detected_, Proportion_>& metrics, BlockSource_ block, Output_* output) {
295 constexpr bool unblocked = std::is_same<BlockSource_, bool>::value;
296 std::fill_n(output, n, 1);
297
298 for (size_t i = 0; i < n; ++i) {
299 auto thresh = [&]() {
300 if constexpr(unblocked) {
301 return host.get_sum();
302 } else {
303 return host.get_sum()[block[i]];
304 }
305 }();
306 output[i] = output[i] && (metrics.sum[i] >= thresh);
307 }
308
309 for (size_t i = 0; i < n; ++i) {
310 auto thresh = [&]() {
311 if constexpr(unblocked) {
312 return host.get_detected();
313 } else {
314 return host.get_detected()[block[i]];
315 }
316 }();
317 output[i] = output[i] && (metrics.detected[i] >= thresh);
318 }
319
320 size_t nsubsets = metrics.subset_proportion.size();
321 for (size_t s = 0; s < nsubsets; ++s) {
322 auto sub = metrics.subset_proportion[s];
323 const auto& sthresh = host.get_subset_proportion()[s];
324 for (size_t i = 0; i < n; ++i) {
325 auto thresh = [&]() {
326 if constexpr(unblocked) {
327 return sthresh;
328 } else {
329 return sthresh[block[i]];
330 }
331 }();
332 output[i] = output[i] && (sub[i] <= thresh);
333 }
334 }
335}
