90template<typename Value_, typename Index_, typename Subset_, typename Sum_, typename Detected_, typename Proportion_>
91void compute_rna_qc_metrics(
92 const tatami::Matrix<Value_, Index_>& mat,
93 const std::vector<Subset_>& subsets,
94 const ComputeRnaQcMetricsBuffers<Sum_, Detected_, Proportion_>& output,
95 const ComputeRnaQcMetricsOptions& options)
96{
97 auto NC = mat.ncol();
98 size_t nsubsets = subsets.size();
99
100 PerCellQcMetricsBuffers<Sum_, Detected_, Value_, Index_> tmp;
101 tmp.sum = output.sum;
102 tmp.detected = output.detected;
103
104 constexpr bool same_type = std::is_same<Sum_, Proportion_>::value;
105 typename std::conditional<same_type, bool, std::vector<std::vector<Sum_> > >::type placeholder_subset;
106
107 if (output.subset_proportion.size()) {
108 // Providing space for the subset sums if they're not the same type.
109 if constexpr(same_type) {
110 tmp.subset_sum = output.subset_proportion;
111 } else {
112 placeholder_subset.resize(nsubsets);
113 tmp.subset_sum.resize(nsubsets);
114 for (size_t s = 0; s < nsubsets; ++s) {
115 auto& b = placeholder_subset[s];
116 b.resize(NC);
117 tmp.subset_sum[s] = b.data();
118 }
119 }
120 }
121
122 PerCellQcMetricsOptions opt;
123 opt.num_threads = options.num_threads;
124 per_cell_qc_metrics(mat, subsets, tmp, opt);
125
126 for (size_t s = 0 ; s < nsubsets; ++s) {
127 auto dest = output.subset_proportion[s];
128 if (dest) {
129 auto src = tmp.subset_sum[s];
130 for (Index_ c = 0; c < NC; ++c) {
131 dest[c] = static_cast<Proportion_>(src[c]) / static_cast<Proportion_>(tmp.sum[c]);
132 }
133 }
134 }
135}

234void rna_populate(Host_& host, size_t n, const ComputeRnaQcMetricsBuffers<Sum_, Detected_, Proportion_>& res, BlockSource_ block, const ComputeRnaQcFiltersOptions& options) {
235 constexpr bool unblocked = std::is_same<BlockSource_, bool>::value;
236 auto buffer = [&]() {
237 if constexpr(unblocked) {
238 return std::vector<Float_>(n);

240 return FindMedianMadWorkspace<Float_, size_t>(n, block);
241 }
242 }();

244 {
245 ChooseFilterThresholdsOptions opts;

248 opts.upper = false;

250 if constexpr(unblocked) {
251 return choose_filter_thresholds(n, res.sum, buffer.data(), opts).lower;
252 } else {

254 }
255 }();
256 }

261 opts.log = true;

268 }
269 }();
270 }

278 host.get_subset_proportion().resize(nsubsets);

292template<class Host_, typename Sum_, typename Detected_, typename Proportion_, typename BlockSource_, typename Output_>
293void rna_filter(const Host_& host, size_t n, const ComputeRnaQcMetricsBuffers<Sum_, Detected_, Proportion_>& metrics, BlockSource_ block, Output_* output) {
294 constexpr bool unblocked = std::is_same<BlockSource_, bool>::value;
295 std::fill_n(output, n, 1);
296
297 for (size_t i = 0; i < n; ++i) {
298 auto thresh = [&]() {
299 if constexpr(unblocked) {
300 return host.get_sum();
301 } else {
302 return host.get_sum()[block[i]];
303 }
304 }();
305 output[i] = output[i] && (metrics.sum[i] >= thresh);
306 }
307
308 for (size_t i = 0; i < n; ++i) {
309 auto thresh = [&]() {
310 if constexpr(unblocked) {
311 return host.get_detected();
312 } else {
313 return host.get_detected()[block[i]];
314 }
315 }();
316 output[i] = output[i] && (metrics.detected[i] >= thresh);
317 }
318
319 size_t nsubsets = metrics.subset_proportion.size();
320 for (size_t s = 0; s < nsubsets; ++s) {
321 auto sub = metrics.subset_proportion[s];
322 const auto& sthresh = host.get_subset_proportion()[s];
323 for (size_t i = 0; i < n; ++i) {
324 auto thresh = [&]() {
325 if constexpr(unblocked) {
326 return sthresh;
327 } else {
328 return sthresh[block[i]];
329 }
330 }();
331 output[i] = output[i] && (sub[i] <= thresh);
332 }
333 }
334}
