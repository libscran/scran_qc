92template<typename Value_, typename Index_, typename Subset_, typename Sum_, typename Detected_, typename Proportion_>
93void compute_rna_qc_metrics(
94 const tatami::Matrix<Value_, Index_>& mat,
95 const std::vector<Subset_>& subsets,
96 const ComputeRnaQcMetricsBuffers<Sum_, Detected_, Proportion_>& output,
97 const ComputeRnaQcMetricsOptions& options)
98{
99 auto NC = mat.ncol();
100 size_t nsubsets = subsets.size();
101
102 PerCellQcMetricsBuffers<Sum_, Detected_, Value_, Index_> tmp;
103 tmp.sum = output.sum;
104 tmp.detected = output.detected;
105
106 constexpr bool same_type = std::is_same<Sum_, Proportion_>::value;
107 typename std::conditional<same_type, bool, std::vector<std::vector<Sum_> > >::type placeholder_subset;
108
109 if (output.subset_proportion.size()) {
110 // Providing space for the subset sums if they're not the same type.
111 if constexpr(same_type) {
112 tmp.subset_sum = output.subset_proportion;
113 } else {
114 placeholder_subset.resize(nsubsets);
115 tmp.subset_sum.resize(nsubsets);
116 for (size_t s = 0; s < nsubsets; ++s) {
117 auto& b = placeholder_subset[s];
118 b.resize(NC);
119 tmp.subset_sum[s] = b.data();
120 }
121 }
122 }
123
124 PerCellQcMetricsOptions opt;
125 opt.num_threads = options.num_threads;
126 per_cell_qc_metrics(mat, subsets, tmp, opt);
127
128 for (size_t s = 0 ; s < nsubsets; ++s) {
129 auto dest = output.subset_proportion[s];
130 if (dest) {
131 auto src = tmp.subset_sum[s];
132 for (Index_ c = 0; c < NC; ++c) {
133 dest[c] = static_cast<Proportion_>(src[c]) / static_cast<Proportion_>(tmp.sum[c]);
134 }
135 }
136 }
137}

235template<typename Float_, class Host_, typename Sum_, typename Detected_, typename Proportion_, typename BlockSource_>
236void rna_populate(Host_& host, size_t n, const ComputeRnaQcMetricsBuffers<Sum_, Detected_, Proportion_>& res, BlockSource_ block, const ComputeRnaQcFiltersOptions& options) {
237 constexpr bool unblocked = std::is_same<BlockSource_, std::false_type>::value;
238 constexpr bool oneblocked = std::is_same<BlockSource_, std::true_type>::value;
239
240 auto buffer = [&]() {
241 if constexpr(unblocked || oneblocked) {
242 return std::vector<Float_>(n);
243 } else {
244 return FindMedianMadWorkspace<Float_, size_t>(n, block);
245 }
246 }();
247
248 {
249 ChooseFilterThresholdsOptions opts;
250 opts.num_mads = options.sum_num_mads;
251 opts.log = true;
252 opts.upper = false;
253 host.get_sum() = [&]() {
254 if constexpr(unblocked || oneblocked) {
255 auto thresh = choose_filter_thresholds(n, res.sum, buffer.data(), opts).lower;
256 if constexpr(unblocked) {
257 return thresh;
258 } else {
259 return std::vector<Float_>(1, thresh);
260 }
261 } else {
262 return internal::strip_threshold<true>(choose_filter_thresholds_blocked(n, res.sum, block, &buffer, opts));
263 }
264 }();
265 }
266
267 {
268 ChooseFilterThresholdsOptions opts;
269 opts.num_mads = options.detected_num_mads;
270 opts.log = true;
271 opts.upper = false;
272 host.get_detected() = [&]() {
273 if constexpr(unblocked || oneblocked) {
274 auto thresh = choose_filter_thresholds(n, res.detected, buffer.data(), opts).lower;
275 return maybe_vector<oneblocked>(thresh);
276 } else {
277 return internal::strip_threshold<true>(choose_filter_thresholds_blocked(n, res.detected, block, &buffer, opts));
278 }
279 }();
280 }
281
282 {
283 ChooseFilterThresholdsOptions opts;
284 opts.num_mads = options.subset_proportion_num_mads;
285 opts.lower = false;
286
287 size_t nsubsets = res.subset_proportion.size();
288 host.get_subset_proportion().resize(nsubsets);
289 for (size_t s = 0; s < nsubsets; ++s) {
290 auto sub = res.subset_proportion[s];
291 host.get_subset_proportion()[s] = [&]() {
292 if constexpr(unblocked || oneblocked) {
293 auto thresh = choose_filter_thresholds(n, sub, buffer.data(), opts).upper;
294 return maybe_vector<oneblocked>(thresh);
295 } else {
296 return internal::strip_threshold<false>(choose_filter_thresholds_blocked(n, sub, block, &buffer, opts));
297 }
298 }();
299 }
300 }

303template<class Host_, typename Sum_, typename Detected_, typename Proportion_, typename BlockSource_, typename Output_>
304void rna_filter(const Host_& host, size_t n, const ComputeRnaQcMetricsBuffers<Sum_, Detected_, Proportion_>& metrics, BlockSource_ block, Output_* output) {
305 std::fill_n(output, n, 1);
306
307 const auto& sum = host.get_sum();
308 for (size_t i = 0; i < n; ++i) {
309 output[i] = output[i] && (metrics.sum[i] >= define_threshold(sum, block, i));
310 }
311
312 const auto& detected = host.get_detected();
313 for (size_t i = 0; i < n; ++i) {
314 output[i] = output[i] && (metrics.detected[i] >= define_threshold(detected, block, i));
315 }
316
317 size_t nsubsets = metrics.subset_proportion.size();
318 for (size_t s = 0; s < nsubsets; ++s) {
319 auto sub = metrics.subset_proportion[s];
320 const auto& sthresh = host.get_subset_proportion()[s];
321 for (size_t i = 0; i < n; ++i) {
322 output[i] = output[i] && (sub[i] <= define_threshold(sthresh, block, i));
323 }
324 }
325}
